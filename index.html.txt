    function renderProducts(filteredProducts) {
      const container = document.getElementById('products');
      container.innerHTML = filteredProducts.map(p => `
        <div class="bg-white rounded-3xl overflow-hidden shadow-lg hover:shadow-2xl transition-all">
          <img src="${p.image}" class="w-full h-64 object-cover">
          <div class="p-6">
            <h3 class="text-xl font-semibold mb-1">${p.name}</h3>
            <p class="text-stone-500 text-sm">${p.diameter} • ${p.capacity}</p>

              Gauti pasiūlymą
            </button>
          </div>
        </div>
      `).join('');
    }

    // Filtravimas ir kategorijos
    function filterProducts() {
      const search = document.getElementById('search').value.toLowerCase();
      const material = document.getElementById('material').value;
      
      let filtered = produktai;

      if (search) {
        filtered = filtered.filter(p => p.name.toLowerCase().includes(search));
      }
      if (material) {
        filtered = filtered.filter(p => p.material === material);
      }

      renderProducts(filtered);
    }

    // Rodyti kategoriją
    function showCategory(cat) {
      document.querySelectorAll('button').forEach(btn => btn.classList.remove('active'));
      
      if (cat === 'all') document.getElementById('btn-all').classList.add('active');
      if (cat === 'kubilai') document.getElementById('btn-kubilai').classList.add('active');
      if (cat === 'pirtys') document.getElementById('btn-pirtys').classList.add('active');

      let filtered = produktai;
      if (cat !== 'all') {
        filtered = produktai.filter(p => p.category === cat);
      }
      renderProducts(filtered);
    }

    // Pradinis užkrovimas + live filtravimas
    document.getElementById('search').addEventListener('input', filterProducts);
    document.getElementById('material').addEventListener('change', filterProducts);
